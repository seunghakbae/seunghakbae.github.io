# 파이썬 코드

~~~python
st = input()

rev = []
word = ""
tag = False

for s in st:
    if s == '<':
        tag = True
        rev.append(word[::-1])
        rev.append(s)
        word = ''
    elif s == '>':
        tag = False
        rev.append(s)
        word = ''
    elif tag:
        rev.append(s)
    else:
        if s != ' ':
            word += s
        else:
            rev.append(word[::-1])
            rev.append(' ')
            word = ""

rev.append(word[::-1])

for i in rev:
    print(i, end='')
~~~

~~~python
S, tmp = input(), ""

ck = False

for i in S:
    if i == ' ':
        if not ck:
            print(tmp[::-1], end=' ')
            tmp = ''
        else:
            print(' ', end="")
    elif i == '<':
        print()
        ck = True
        print(tmp[::-1] + '<', end='')
        tmp = ""
    elif i == '>':
        ck = False
        print('>', end='')
    else:
        if ck:
            print(i, end='')
        else:
            tmp += i

print(tmp[::-1])
~~~
